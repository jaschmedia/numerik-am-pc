\subsubsection*{midpoint.m}
\begin{lstlisting}[frame=single]
function [ I ] = midpoint( f,a,b,N )
h=(b-a)/N;
x=a:h:b;
x=x(1:end-1)+h/2;
fx=f(x);
I=h*sum(fx);
end
\end{lstlisting}

\subsubsection*{trapez.m}
\begin{lstlisting}[frame=single]
function [ I ] = trapez( f,a,b,N )
h=(b-a)/N;
x=a:h:b;
fx=f(x);
I=(h/2)*(fx(1)+fx(end))+h*sum(fx(2:end-1));
end
\end{lstlisting}
